using System.Collections;
using UnityEngine;
using UnityEngine.AI;

public class enemyAI : MonoBehaviour, iDamage
{
    // === Visuals & Components ===
    [SerializeField] Renderer model;        // Our stylish outer shell – used for flashing red on hit
    [SerializeField] NavMeshAgent agent;    // Our pathfinding brain – handles the chase!

    // === Damage & HP ===
    [SerializeField] int HP;                // Our health – keep it above zero or we’re toast
    [SerializeField] float shootRate;       // How often we get to go pew pew

    // === Combat Stuff ===
    [SerializeField] Transform shootPos;    // Where the bullets spawn – basically our muzzle
    [SerializeField] GameObject bullet;     // The actual bullet prefab – bringin’ the pain

    // === Rotation & Targeting ===
    [SerializeField] int faceTargetSpeed;   // How snappy our head-turns are – no Exorcist spins

    // === Runtime Goodies ===
    float shootTimer;            // Keeping track of how long since the last shot
    Color colorOrig;             // Our original color, so we can bounce back after flashing red
    Vector3 playerDir;           // Direction to the player – useful for rotating like a boss

    // Called once when the enemy spawns in
    void Start()
    {
        colorOrig = model.material.color;          // Store our default color to reset after damage
        gameManager.instance.updateGameGoal(1);    // Let the game know we’ve joined the party
    }

    // Runs every frame – the main event loop for our baddie
    void Update()
    {
        // Track the player’s position so we know where to go (and look menacing)
        playerDir = (gameManager.instance.player.transform.position - transform.position);

        // Set our destination to the player – full stalker mode
        agent.SetDestination(gameManager.instance.player.transform.position);

        // If we’ve reached our target (kinda), time to face them dramatically
        if (agent.remainingDistance <= agent.stoppingDistance)
        {
            FaceTarget();
        }

        // Tick up the timer to know when we can shoot again
        shootTimer += Time.deltaTime;

        // If cooldown's done, unleash the bullet
        if (shootTimer >= shootRate)
        {
            Shoot();
        }
    }

    // Interface call – damage incoming!
    public void TakeDamage(int amount)
    {
        HP -= amount;              // Ow. That hurt.
        StartCoroutine(FlashRed()); // Let’s let everyone *see* how much that hurt

        // If health drops to 0 or below, it's curtains for us
        if (HP <= 0)
        {
            gameManager.instance.updateGameGoal(-1); // Let the game know we checked out
            Destroy(gameObject);                     // Bye bye, cruel world
        }
    }

    // Quick red flash when we get smacked – visual feedback FTW
    IEnumerator FlashRed()
    {
        model.material.color = Color.red;            // Angry tomato mode
        yield return new WaitForSeconds(0.1f);        // Hold it for dramatic effect
        model.material.color = colorOrig;            // Back to chill
    }

    // Pew pew – spawn a bullet and let physics do the rest
    void Shoot()
    {
        shootTimer = 0;                                      // Reset the cooldown
        Instantiate(bullet, shootPos.position, transform.rotation); // Fire away!
    }

    // Turn to face the player like you're about to drop the villain monologue
    void FaceTarget()
    {
        Quaternion rot = Quaternion.LookRotation(new Vector3(playerDir.x, transform.position.y, playerDir.z));
        transform.rotation = Quaternion.Lerp(transform.rotation, rot, Time.deltaTime * faceTargetSpeed);
    }

    // Required by iDamage but not used here – throw an exception if someone tries
    public void takeDamage(int amount)
    {
        throw new System.NotImplementedException(); // Not implemented... yet?
    }
}

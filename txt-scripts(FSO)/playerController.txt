using UnityEngine;

public class playerController : MonoBehaviour, iDamage
{
    // === Player Speed & Movement ===
    [SerializeField] int speed;          // Base speed – your regular ol’ jogging pace
    [SerializeField] int sprintMod;      // Sprint multiplier – kick it into overdrive!
    [SerializeField] int jumpVelocity;   // How much we launch into the sky when jumping
    [SerializeField] int jumpMax;        // Max number of jumps before gravity wins
    [SerializeField] CharacterController controller; // Our ride through Unity’s physics

    // === Gravity ===
    [SerializeField] int gravity;        // Gravity’s pull – like a clingy ex, but for real

    // === Damage & HP ===
    [SerializeField] int HP;             // How much life we got left – don’t let it hit 0!
    [SerializeField] int shootDamage;    // Pew pew power – damage dealt per shot
    [SerializeField] int shootDist;      // How far we can shoot – no sniping across the map
    [SerializeField] float shootRate;    // Cooldown between shots – no spamming!

    // === Miscellaneous ===
    [SerializeField] LayerMask ignoreLayer; // What we’re NOT allowed to shoot – friendlies or decor

    // === Runtime Vibes ===
    float shootTimer;        // Keeps track of time since last shot
    int jumpCount;           // How many times we’ve jumped without touching grass
    int origHP;              // Our starting HP – just in case we need a reset
    Vector3 moveDirection;   // Direction we're moving in this frame
    Vector3 playerVelocity;  // For storing vertical velocity mostly
    bool isSprinting;        // Currently sprinting? Who knows. Let’s track it!

    void Start()
    {
        origHP = HP; // Save that sweet original HP in case we ever want to respawn or something
    }

    void Update()
    {
        // Just drawing a big ol’ red laser from the camera so we can see where we’re aiming
        Debug.DrawRay(Camera.main.transform.position, Camera.main.transform.forward * shootDist, Color.red);

        // Movement magic happens here
        movement();

        // Gotta go fast? This checks if we're sprinting
        sprint();
    }

    void movement()
    {
        // If we’re grounded, reset jumps and vertical velocity – thanks gravity!
        if (controller.isGrounded)
        {
            jumpCount = 0;
            playerVelocity = Vector3.zero;
        }

        // Getting movement input and translating it relative to our character's direction
        moveDirection = (Input.GetAxis("Horizontal") * transform.right) +
                        (Input.GetAxis("Vertical") * transform.forward);

        // Actually move the character based on our direction and speed
        controller.Move(moveDirection * speed * Time.deltaTime);

        // Attempt to jump if the button’s pressed and we still got jumps left
        jump();

        // Gravity keeps pulling us down – every frame
        playerVelocity.y -= gravity * Time.deltaTime;

        // Apply our vertical movement (falling, jumping) to the character
        controller.Move(playerVelocity * speed * Time.deltaTime);

        // Keep track of time between shots
        shootTimer += Time.deltaTime;

        // If fire button is held and cooldown is done, fire away!
        if (Input.GetButton("Fire1") && shootTimer >= shootRate)
        {
            shoot();
        }
    }

    void jump()
    {
        // If player hits jump and we haven’t maxed out our jumps, launch upward!
        if (Input.GetButtonDown("Jump") && jumpCount < jumpMax)
        {
            jumpCount++;
            playerVelocity.y = jumpVelocity;
        }
    }

    void sprint()
    {
        // Sprint button down? Time to double that hustle
        if (Input.GetButtonDown("Sprint"))
        {
            speed *= sprintMod;
        }
        // Sprint button released? Back to your usual tired jog
        else if (Input.GetButtonUp("Sprint"))
        {
            speed /= sprintMod;
        }
    }

    void shoot()
    {
        shootTimer = 0; // Reset cooldown – time for another bang!

        RaycastHit hit;
        // Pew pew – shoot a ray forward from the camera, ignoring specified layers
        if (Physics.Raycast(Camera.main.transform.position, Camera.main.transform.forward, out hit, shootDist, ~ignoreLayer))
        {
            Debug.Log(hit.collider.name); // Just so we know who we hit – hello there!

            // Check if we hit something damageable
            iDamage dmg = hit.collider.GetComponent<iDamage>();

            if (dmg != null)
            {
                // Apply the pain
                dmg.takeDamage(shootDamage);
            }
        }
    }

    public void takeDamage(int amount)
    {
        HP -= amount; // Ouch. We lost some health.

        // If we’ve run out of HP, it’s game over, man!
        if (HP <= 0)
        {
            gameManager.instance.youLose(); // Let the GameManager handle our dramatic demise
        }
    }
}
